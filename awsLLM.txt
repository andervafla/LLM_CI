  }
}

  name     = "main-tg"
  port     = 80
  protocol = "HTTP"

  health_check {
    path                = "/"
    interval            = 30
    timeout             = 5
    healthy_threshold   = 2
    unhealthy_threshold = 2
  }
}

  default_action {

    target_group_arn = aws_lb_target_group.tg.arn
  }
}
